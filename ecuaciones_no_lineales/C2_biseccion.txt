Pseudocódigo para el algoritmo correspondiente al método de biseccion en el cual se buscan raíces de una ecuación 
a partir de un intervalo dado.

Data: f (función), Xi (valor izquierdo), Xf (valor derecho), tol (tolerancia), i (iteraciones)
Result: raíz sin error r, raíz con error(aproximado) ri  o mensaje de fallo

Leer Xi, Xf, tol, i 
Yi <- f(Xi)
Yf <- f(Xf)
if Yi = 0 then
    Mostrar Xi es raíz
else 
    if Yf = 0 then
        Mostrar Xf es raíz
    else
        Xm <- (Xi + Xf) / 2
        Ym <- f(Xm)
        error <- tol + 1
        contador <- 1
        while Ym != 0 ∧ error > tol ∧ contador < i do 
            if Yi * Ym > 0 then
                Xi <- Xm
                Yi <- Ym
            else 
                Xaux <- Xm
                Xm <- (Xi + Xf) / 2
                Ym <- f(Xm)
                error <- |Xm - Xaux| v |(Xm - Xaux)/Xm|
                contador <- contador + 1
        if Ym = 0 then
            Mostrar Xm es raíz
        else
            if error <= tol then
                Mostrar Xm es una raíz con un error igual a error
            else 
                Mostrar El metodo falló
